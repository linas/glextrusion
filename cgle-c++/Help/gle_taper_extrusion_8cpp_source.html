49 void CgleTaperExtrusion::Draw(double Length, double Distortion,
50  float Color_Array[][3], bool bTextured)
51 {
52 
53  double startz = -(Length / 2.0);
54  double endz = Length / 2.0;
55  double si, co, taper, aponent;
56  double taper_step, taper_z;
57 
58 
59  double currz, delta;
60  double currang, delang;
61  aponent = fabs(Distortion);
62 
63  if (aponent == 0.0)
64  aponent = 1.0E-10;
65  taper_z = -.9999;
66  taper_step = 1.9998 / (m_iPoints - 3);
67 
68  delta = (endz - startz) / ((double)(m_iPoints - 3));
69  currz = startz - delta;
70  delang = 2.0 * (FTwist / ((double)(m_iPoints - 3)));
71  currang = -delang;
72 
73  for (int i = 1; i< m_iPoints - 1; i++)
74  {
75  m_ptrPath[i][0] = 0.0;
76  m_ptrPath[i][1] = 0.0;
77  m_ptrPath[i][2] = currz;
78 
79  si = sin(currang);
80  co = cos(currang);
81  taper = pow((1.0 - pow(fabs(taper_z), 1.0 / aponent)), aponent);
82  taper_z += taper_step;
83 
84  m_ptrXforms[i][0][0] = taper * co;
85  m_ptrXforms[i][0][1] = -taper * si;
86  m_ptrXforms[i][0][2] = 0.0;
87  m_ptrXforms[i][1][0] = taper * si;
88  m_ptrXforms[i][1][1] = taper * co;
89  m_ptrXforms[i][1][2] = 0.0;
90 
91  currz += delta;
92  currang += delang;
93  }
94 
95  m_ptrPath[0][0] = m_ptrPath[1][0];
96  m_ptrPath[0][1] = m_ptrPath[1][1];
97  m_ptrPath[0][2] = m_ptrPath[1][2];

gleTexMode::GLE_TEXTURE_NORMAL_MODEL_CYL
Similar to GLE_TEXTURE_NORMAL_CYL, except that the untransformed normals are used. As a result, textures tends to stick to the extrusion according to the extrusions local surface coordinates rather than according to real-space coordinates. This will in general provide the correct style of texture mapping when affine transforms are being applied to the contour, since the coordinates used are those prior to the affine transform.
